/* for Insert*/
CREATE PROCEDURE Insert_Student
	@ID int,
	@Name nvarchar(30),
	@DOB nvarchar(50),
	@Add nvarchar(30),
	@Gen nvarchar(10),
	@Mob decimal(10)
AS
begin
Insert into CRUD_Application (Student_ID,Name,Date_Of_Birth,Address,Gender,Mobile_No) 
Values(@ID,@Name,@DOB,@Add,@Gen,@Mob);
end

/* for View */

CREATE PROCEDURE View_Student	
AS
begin
Select * from CRUD_Application ;
end
exec View_Student

/* for Update */
CREATE PROCEDURE Update_Student
	@ID int,
	@Name nvarchar(30),
	@DOB nvarchar(50),
	@Add nvarchar(30),
	@Gen nvarchar(10),
	@Mob decimal(10)
AS
begin
Update CRUD_Application set Name = @Name,Date_Of_Birth = @DOB,Address = @Add,Gender = @Gen,Mobile_No = @Mob where Student_ID = @ID ;
end

/* for Delete */
CREATE PROCEDURE Delete_Student
	@ID int
AS
begin
Delete CRUD_Application where Student_ID = @ID ;
end

/* for Search */
CREATE PROCEDURE Search_Student
	@ID int
AS
begin
Select * from CRUD_Application where Student_ID = @ID ;
end
